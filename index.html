Task 1. Для данного n возьмите сумму цифр n. Если это значение имеет более одной цифры, продолжайте уменьшать таким образом, пока не будет получено однозначное число. Входными данными будет неотрицательное целое число.

            let digital_root = num => 
            num > 9 ?
             digital_root([...num.toString()]
            .reduce((prev, item) => +prev + +item)) : num;
          
  
            //digital_root(16)  =>   7
            //digital_root(942)  =>   6
            //digital_root(132189)  =>  6
            //digital_root(493193)  =>   2

            let uniqueInOrder = str => (typeof str == 'number')?
            (str.toString().replace(/(.)\1*/g, '$1')

            str.replace(/(.)\1*/g, '$1').split('')

            let  validParentheses = str =>{
               let array = str;
               while (array.includes("()")){
                 array = array.split("()").join("")
                 }
                 return array.length === 0 ? true : false
               }

Task 8. Если мы перечислим все натуральные числа ниже 10, которые кратны 3 или 5, мы получим 3, 5, 6 и 9. Сумма этих кратных 23. Завершите решение так, чтобы оно возвращало сумму всех кратных 3 или 5 числам ниже переданного числа.
            
               let result =[];
               let solution = num =>{
               for (let i=1; i<num; i++){
                   if ((i%3==0)||(i%5==0)){
                       result.push(i)
                   }
                 }
                 return result.reduce((prev,item)=>prev+item,0)}
          
 
            //solution(10) => 23
            //solution(12) => 33
            //solution(23) => 119
Task 9. Посчитать лайки на фейсбуке
  
               let likes = arr => {
               let arrLength = arr.length;
               let count = arrLength-2;
               if ((arrLength==0)){
                   return `no one likes this`
               }
               let result = arr.map((item,index,array)=>{
                return (arrLength==1)?(`${item} likes this`):
                (arrLength==2)?(`${array[0]} and ${array[1]} like this`):
                (arrLength==3)?(`${array[0]}, ${array[1]} and ${array[2]} like this`):
                (`${array[0]} and ${arr[1]} and ${count} others like this`)
               })
               return result.filter((item, pos) => result.indexOf(item) == pos).join('')
            
  
            // likes([]) => 'no one likes this'
            // likes(['Peter']) => 'Peter likes this'
            // likes(['Jacob', 'Alex']) => 'Jacob and Alex like this'
            // likes(['Max', 'John', 'Mark']) => 'Max, John and Mark like this')
            // likes(['Alex', 'Jacob', 'Mark', 'Max']) => 'Alex, Jacob and 2 others like this'
Task 10. Нарциссическое число - это число длины n, в котором сумма цифр в степени n равна исходному числу.

               let isNarcissistic = num =>  num.toString()
               .split('')
               .reduce((prev,item,index,array)=>
               prev+(+item)**array.length,0) == num
               
  
              // isNarcissistic(153) => true;
              // isNarcissistic(435) => false;
              // isNarcissistic(370) => true;
